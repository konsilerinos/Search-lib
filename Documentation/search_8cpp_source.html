172void Search::Sort(std::vector<int>& array, int start, int end) {
173
174 // Сортировка не требуется
175 if (end - start < 2) {
176
177 return;
178 // Пара элементов, требуется сортировка
179 } else if (end - start == 2) {
180
181 // Установить порядок по возрастанию
182 if (array[start] > array[start + 1]) {
183
184 Swap(array[start], array[start + 1]);
185 }
186 }
187
188 // Вектор. Используется для слияния сортируемых частей
189 std::vector<int> temp(end - start, 0);
190
191 // Деление вектора на две части
192 Sort(array, start, (start + end) / 2);
193 Sort(array, (start + end) / 2, end);
194
195 // Сохранение границ первой части
196 int start1 = start;
197 int end1 = (start + end) / 2;
198
199 // Сохранение границ второй части
200 int start2 = end1;
201 int end2 = end;
202
203 // Слияние сортируемых частей
204 for (int i = 0; i < end - start; i++) {
205
206 // Выход за границы первой части или элемент второй части меньша элемента первой
207 // Сохранение в temp элементов второй части
208 if ((start1 >= end1) || ((start2 < end2) && (array[start2] < array[start1]))) {
209
210 temp[i] = array[start2];
211 start2++;
212 // Сохранение в temp элементов первой части
213 } else {
214
215 temp[i] = array[start1];
216 start1++;
217 }
218 }
219
220 // Сохранение отсортированного отрезка [start, end] в исходный массив
221 for (int i = start; i < end; i++) {
222 array[i] = temp[i - start];
223 }
224
225 // Очистка вектора
226 temp.clear();
227}

317IterTime Search::BinarySearch(std::vector<int>& array, int value) {
318
319 // Точка времени, начало
320 std::chrono::steady_clock::time_point beginTime = std::chrono::steady_clock::now();
321
322 // Левая граница поиска
323 int left = 0;
324 // Правая граница поиска
325 int right = array.size() - 1;
326 // Середина отрезка [left; right]
327 int middle = 0;
328
329 // Итератор либо на искомый элемент, либо на конец вектора
330 std::vector<int>::iterator iter = array.begin();
331
332 // Если произошло пересечение отрезков - поиск завершён
333 while (right - left > 1) {
334
335 // Обновление середины
336 middle = (left + right) / 2;
337
338 // Сравнение элемента по середине с искомым
339 if (array[middle] < value) {
340
341 // Сместить левую границу вправо
342 left = middle;
343 } else {
344
345 // Сместить правую границу влево
346 right = middle;
347 }
348 }
349
350 // Точка времени, конец
351 std::chrono::steady_clock::time_point endTime = std::chrono::steady_clock::now();
352
353 // Наносекунды, время выполнения
354 std::chrono::nanoseconds resultTime = std::chrono::duration_cast<std::chrono::nanoseconds>(endTime - beginTime);
355
356 // Проверка на попадание элемента по левой границе
357 if (array[left] == value) {
358
359 return {resultTime, iter + left};
360 }
361
362 // Проверка на попадание элемента по правой границе
363 if (array[right] == value) {
364
365 return {resultTime, iter + right};
366 }
367
368 // Элемент не найден
369 return {resultTime, array.end()};
370}
